158  @Override
159  public final IFuzzyValue<Boolean> execute( final IContext p_context, final boolean p_parallel, final List<ITerm> p_argument, final List<ITerm> p_return,
160  final List<ITerm> p_annotation
161  )
162  {
163  final List<ITerm> l_arguments = CCommon.flatcollection( p_argument ).collect( Collectors.toList() );
164 
165  IntStream.range( 0, l_arguments.size() )
166  .filter( i -> CCommon.rawvalueAssignableTo( l_arguments.get( i ), String.class ) )
167  .mapToObj( i -> new AbstractMap.SimpleImmutableEntry<>( i, l_arguments.get( i ).<String>raw() ) )
168  .filter( i -> EDistribution.exist( i.getValue() ) )
169  .map( i -> new AbstractMap.SimpleImmutableEntry<>( i.getKey(), EDistribution.from( i.getValue() ) ) )
170  .map( i -> {
171 
172  // check if next argument to the distribution name a generator name
173  final int l_skip;
174  final EGenerator l_generator;
175 
176  if ( ( i.getKey() < l_arguments.size() - 1 ) && ( CCommon.rawvalueAssignableTo( l_arguments.get( i.getKey() + 1 ), String.class ) ) )
177  {
178  l_skip = 1;
179  l_generator = EGenerator.from( l_arguments.get( i.getKey() + 1 ).<String>raw() );
180  }
181  else
182  {
183  l_skip = 0;
184  l_generator = EGenerator.MERSENNETWISTER;
185  }
186 
187  // generate distribution object, arguments after distribution are the initialize parameter
188  return i.getValue()
189  .get(
190  l_generator.get(),
191  l_arguments.stream()
192  .skip( i.getKey() + 1 + l_skip )
193  .limit( i.getValue().getArgumentNumber() )
194  .map( ITerm::<Number>raw )
195  .mapToDouble( Number::doubleValue )
196  .toArray()
197  );
198 
199  } )
200  .map( CRawTerm::from )
201  .forEach( p_return::add );
202 
203  return CFuzzyValue.from( true );
204  }

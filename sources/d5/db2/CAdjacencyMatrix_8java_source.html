144  private static Pair<DoubleMatrix2D, Collection<?>> apply( final AbstractGraph<Object, Object> p_graph, final Map<?, Number> p_cost,
145  final double p_defaultcost, final EType p_type )
146  {
147  // index map for matching vertex to index position within matrix
148  final Map<Object, Integer> l_index = new HashMap<>();
149 
150  // extract vertices from edges
151  p_graph.getEdges()
152  .stream()
153  .map( p_graph::getEndpoints )
154  .flatMap( i -> Stream.of( i.getFirst(), i.getSecond() ) )
155  .forEach( i -> l_index.putIfAbsent( i, 0 ) );
156 
157  // define for each vertex an index number in [0, size)
158  StreamUtils.zip(
159  l_index.keySet().stream(),
160  IntStream.range( 0, l_index.size() ).boxed(),
161  ImmutablePair::new
162  ).forEach( i -> l_index.put( i.getKey(), i.getValue() ) );
163 
164  final DoubleMatrix2D l_matrix;
165  switch ( p_type )
166  {
167  case SPARSE:
168  l_matrix = new SparseDoubleMatrix2D( l_index.size(), l_index.size() );
169  break;
170 
171  default:
172  l_matrix = new DenseDoubleMatrix2D( l_index.size(), l_index.size() );
173  }
174 
175  // map costs to matrix
176  p_graph.getEdges()
177  .stream()
178  .map( i -> new ImmutablePair<>( p_graph.getEndpoints( i ), p_cost.getOrDefault( i, p_defaultcost ).doubleValue() ) )
179  .forEach( i -> l_matrix.setQuick(
180  l_index.get( i.getLeft().getFirst() ), l_index.get( i.getLeft().getSecond() ),
181  i.getRight() + l_matrix.getQuick( l_index.get( i.getLeft().getFirst() ), l_index.get( i.getLeft().getSecond() ) )
182  ) );
183 
184  return new ImmutablePair<>( l_matrix, new ArrayList<>( l_index.keySet() ) );
185  }

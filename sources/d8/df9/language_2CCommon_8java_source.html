132  {
133  // filter for avoid duplicated instantiation on non-existing values / annotations

345  public static double levenshtein( final String p_first, final String p_second, final double p_insertweight, final double p_replaceweight,
346  final double p_deleteweight )
347  {
348  // the array of distances
349  double[] l_cost = IntStream.range( 0, p_first.length() + 1 ).mapToDouble( i -> i ).toArray();
350  double[] l_newcost = new double[l_cost.length];
351 
352  for ( int j = 1; j < p_second.length() + 1; j++ )
353  {
354  l_newcost[0] = j;
355 
356  // calculate cost of operation for all characters
357  for ( int i = 1; i < l_cost.length; i++ )
358  l_newcost[i] = min(
359  l_cost[i - 1] + ( p_first.charAt( i - 1 ) == p_second.charAt( j - 1 ) ? 0 : p_replaceweight ),
360  l_newcost[i - 1] + p_deleteweight,
361  l_cost[i] + p_insertweight
362  );
363 
364  final double[] l_swap = l_cost;
365  l_cost = l_newcost;
366  l_newcost = l_swap;
367  }
368 
369  return l_cost[p_first.length()];
370  }

395  public static double ncd( final ECompression p_compression, final String p_first, final String p_second )
396  {
397  final double l_first = compress( p_compression, p_first );
398  final double l_second = compress( p_compression, p_second );
399  return ( compress( p_compression, p_first + p_second ) - Math.min( l_first, l_second ) ) / Math.max( l_first, l_second );
400  }

411  private static double compress( final ECompression p_compression, final String p_input )
412  {
413  final DataOutputStream l_counting = new DataOutputStream( new NullOutputStream() );
414 
415  try (
416  final InputStream l_input = new ByteArrayInputStream( p_input.getBytes( StandardCharsets.UTF_8 ) );
417  final OutputStream l_compress = p_compression.get( l_counting )
